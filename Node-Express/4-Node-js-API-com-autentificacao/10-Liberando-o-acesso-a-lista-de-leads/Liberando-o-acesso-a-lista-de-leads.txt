Código 1. Código servico_leads.js


import pool from "../conexao/conexao.js";

export async function RetornaLeads() {
    const conexao = await pool.getConnection();

    const resposta = await conexao.query("SELECT id, nome, email, telefone FROM cadastro_leads");
    conexao.release();

    const leads = resposta[0];

    return leads;
}
