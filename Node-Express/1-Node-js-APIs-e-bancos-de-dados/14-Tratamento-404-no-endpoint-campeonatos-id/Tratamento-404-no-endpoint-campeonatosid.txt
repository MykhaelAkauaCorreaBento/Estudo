app.get('/campeonatos/:id', async (req, res) => {
    const id = req.params.id;

    const campeonato = await retornaCampeonatosID(id);

    if(campeonato.length > 0) {
        res.json(campeonato);
    } else {
        res.status(404).json({mensagem: "Nenhum campeonato encontrado"})
    }
});
