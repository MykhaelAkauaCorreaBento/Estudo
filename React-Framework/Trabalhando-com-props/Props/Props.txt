import estilos from './page.module.css';

Além dessa, existe outra forma de receber dados conhecida como desestruturação.
Veja um exemplo abaixo:

import estilos from './CardProduto.module.css';
import Image from 'next/image';

export default function CardProduto({ imagem, nome, preco, categoria }) {
  return (
    <div className={estilos.card_container}>
      <div>
        <Image src={imagem} alt={nome}/>
      </div>
      <div>
        <h3>{nome}</h3>
        <p>R${preco}</p>
        <small>{categoria}</small>
        <button>Adicionar</button>
      </div>
    </div>
  );
}

Já a segunda mudança fica na forma como os dados recebidos são inseridos no componente. Veja abaixo:



Forma tradicional de aplicar os dados via props:

export default function CardProduto (props) {
    return (
        <div className={estilos.card_container}>
         <div>
            <h3>{props.nome}<</h3>
            <p>R${props.preco}<</p>
            <small>{props.categoria}</small>
            <button>Adicionar</button>
         </div>
        </div>
    );
}




Formal alternatica de aplicar os dado recebidos via props:

export default function CardProduto({ imagem, nome, preco, categoria }) {
  return (
    <div className={estilos.card_container}>
      <div>
        <Image src={imagem} alt={nome}/>
      </div>
      <div>
        <h3>{nome}</h3>
        <p>R${preco}</p>
        <small>{categoria}</small>
        <button>Adicionar</button>
      </div>
    </div>
  );
}
